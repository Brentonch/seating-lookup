      <button id="searchBtn">Search</button>
      <div class="hint">Tip: Use your official English name format (first name + last name).</div>
      <div id="error" class="notfound"></div>
    </div>

    <div id="result" style="display:none;">
      <div class="field name" id="rName"></div>
      <div class="field"><strong>Department:</strong> <span id="rDept"></span></div>
      <div class="field"><strong>Group:</strong> <span id="rGroup"></span></div>
      <div class="field"><strong>Table:</strong> <span id="rTable"></span></div>
      <button id="backBtn" style="margin-top:12px;background:#444;">Back</button>
    </div>
  </div>

  <script>
    const toKey = s => (s || '').trim().toLowerCase();
    let data = [];

    async function loadData() {
      try {
        const res = await fetch('./data/employees.json', { cache: 'no-store' });
        data = await res.json();
      } catch (e) {
        document.getElementById('error').textContent = 'Failed to load data. Please refresh or contact staff.';
        document.getElementById('searchBtn').disabled = true;
      }
    }

    function findByName(name) {
      const key = toKey(name);
      // Exact match first, then startsWith for tolerance
      let item = data.find(x => toKey(x.englishName) === key);
      if (!item) item = data.find(x => toKey(x.englishName).startsWith(key));
      return item;
    }

    function show(item) {
      document.getElementById('rName').textContent = item.englishName;
      document.getElementById('rDept').textContent = item.department;
      document.getElementById('rGroup').textContent = item.group;
      document.getElementById('rTable').textContent = item.tableLocation;
      document.getElementById('searchSection').style.display = 'none';
      document.getElementById('result').style.display = 'block';
    }

    function back() {
      document.getElementById('result').style.display = 'none';
      document.getElementById('searchSection').style.display = 'block';
      document.getElementById('error').textContent = '';
      document.getElementById('nameInput').focus();
    }

    async function init() {
      await loadData();
      document.getElementById('searchBtn').onclick = () => {
        const inputName = document.getElementById('nameInput').value;
        const item = findByName(inputName);
        if (item) show(item);
        else document.getElementById('error').textContent = 'Name not found. Please check spelling or try full name.';
      };
      document.getElementById('backBtn').onclick = back;
      document.getElementById('nameInput').addEventListener('keyup', (e) => {
        if (e.key === 'Enter') document.getElementById('searchBtn').click();
      });
    }
    init();
  </script>
